Create table CUSTOMER_DETAILS (
CUSTOMER_ID NUMBER(20) PRIMARY KEY,
CUSTOMER_NAME VARCHAR2(255) NOT NULL, 
DOB VARCHAR2(255) NOT NULL ,
GENDER VARCHAR2(10) NOT NULL ,
CONTACT_NUMBER NUMBER(20) NOT NULL,
OCCUPATION VARCHAR2(255) NOT NULL,
ANNUAL_INCOME NUMBER(20) ,
EMAIL_ID VARCHAR2(255) NOT NULL,
RESIDENTIAL_ADDRESS_ID NUMBER(20) NOT NULL, 
PERMANENT_ADDRESS_ID NUMBER(20) NOT NULL 
);

CREATE TABLE ADDRESS_DETAILS(
ADDRESS_ID number(20) PRIMARY KEY, 
ADDRESS_LINE_1 VARCHAR2(255) NOT NULL, 
ADDRESS_LINE_2 VARCHAR2(255),
LANDMARK VARCHAR2(30),
CITY VARCHAR2(30) NOT NULL,
STATE VARCHAR2(30) NOT NULL,
COUNTRY VARCHAR2(20) NOT NULL,
PINCODE number(6) NOT NULL 
);

CREATE TABLE ACCOUNT_DETAILS(
ACCOUNT_NUMBER NUMBER(20) PRIMARY KEY,  
CUSTOMER_ID NUMBER(20) NOT NULL ,
ACCOUNT_TYPE VARCHAR2(30) NOT NULL,
AVAILABLE_BALANCE NUMBER(20) NOT NULL,
IFSC VARCHAR2(30) NOT NULL
);

CREATE TABLE BRANCH_DETAILS(
IFSC VARCHAR2(30) PRIMARY KEY, 
ADDRESS_ID NUMBER(20) NOT NULL,
BRANCH_NAME VARCHAR2(30) NOT NULL
);

CREATE TABLE LOGIN_CREDENTIALS(
CUSTOMER_ID NUMBER(20) PRIMARY KEY,
LOGIN_PASSWORD VARCHAR2(30) NOT NULL,
TRANSACTION_PIN NUMBER(20) NOT NULL 
);

CREATE TABLE IDENTITY_DOCUMENT(
CUSTOMER_ID NUMBER(20) PRIMARY KEY,
PAN VARCHAR2(10) NOT NULL,
AADHAAR_NUMBER NUMBER(12) NOT NULL 
);

Create table VERIFICATION_STATUS(
REFERENCE_ID NUMBER(20) PRIMARY KEY,
CUSTOMER_ID NUMBER(20) NOT NULL, 
VERIFIED CHAR(1) NOT NULL
);

create table TRANSACTION_DETAILS(
TRANSACTION_ID NUMBER(20) PRIMARY KEY,
BENEFACTOR_ACC_NO NUMBER(20) NOT NULL,
BENEFICIARY_ACC_NO NUMBER(20),
BENEFICIARY_ACCOUNT_INFO VARCHAR2(255) NOT NULL,
AMOUNT NUMBER(20) NOT NULL,
DATE_OF_TRANSACTION VARCHAR2(20) NOT NULL,
MODE_OF_TRANSACTION VARCHAR2(20) NOT NULL,
STATUS VARCHAR2(10) NOT NULL,
REMARKS VARCHAR2(255)
);

Create table customer_transaction_statement(
S_NO NUMBER(30) PRIMARY KEY,
CUSTOMER_ID NUMBER(20) NOT NULL,
TRANSACTION_ID NUMBER(20) NOT NULL,
CLOSING_BALANCE NUMBER(20) NOT NULL,
ACTION NUMBER(2) NOT NULL
);

create sequence S_No_Sequence increment by 1 start with 1;
create sequence CustomerId_Sequence increment by 1 start with 10000001;
create sequence AddressId_Sequence increment by 1 start with 1001;
create sequence Account_Number_Sequence increment by 1 start with 100000000001;
create sequence ReferenceId_Sequence increment by 19 start with 1009074030230871;
create sequence Transaction_Id_Sequence increment by 1 start with 1034574030290886;
